#!/usr/bin/env node // Milovi Cake — сборка мини-файлов // Запуск: node build.js const fs = require('fs'); const path = require('path'); function minifyCSS(src) { return src .replace(/\/\*[\s\S]*?\*\//g, '') .replace(/\s+/g, ' ') .replace(/\s*{\s*/g, '{') .replace(/\s*}\s*/g, '}') .replace(/\s*:\s*/g, ':') .replace(/\s*;\s*/g, ';') .replace(/\s*,\s*/g, ',') .replace(/;}/g, '}') .trim(); } function minifyJS(src) { return src .replace(/\/\*[\s\S]*?\*\//g, '') .replace(/^[ \t]+/gm, '') .replace(/[ \t]+$/gm, '') .replace(/\n{2,}/g, '\n') .trim(); } function build(srcPath, dstPath, minFn, label) { const src = fs.readFileSync(srcPath, 'utf8'); const min = minFn(src); fs.writeFileSync(dstPath, min); const from = Math.round(src.length / 1024); const to = Math.round(min.length / 1024); const save = Math.round((1 - min.length / src.length) * 100); console.log(`${label}: ${from}kb → ${to}kb (−${save}%)`); } console.log('\nMilovi Cake build\n'); build('css/style.css', 'css/style.min.css', minifyCSS, 'style.css'); build('js/main.js', 'js/main.min.js', minifyJS, 'main.js '); // Bump DEPLOY_VERSION in sw.js const swPath = 'sw.js'; let sw = fs.readFileSync(swPath, 'utf8'); const today = new Date().toISOString().slice(0, 10); const match = sw.match(/DEPLOY_VERSION = '(\d{4}-\d{2}-\d{2})([a-z]?)'/); if (match) { let newVer; if (match[1] === today) { const next = match[2] ? String.fromCharCode(match[2].charCodeAt(0) + 1) : 'b'; newVer = today + next; } else { newVer = today + 'a'; } sw = sw.replace(/DEPLOY_VERSION = '[^']+'/, `DEPLOY_VERSION = '${newVer}'`); fs.writeFileSync(swPath, sw); console.log(`sw.js: версия → ${newVer}`); } console.log('\nГотово! Задеплой: css/style.min.css, js/main.min.js, sw.js\n');
